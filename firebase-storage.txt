rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    
    // Helper functions
    function isAuthenticated() {
      return request.auth != null;
    }
    
    // Assuming admins will be set with a custom claim `admin: true` in Firebase Auth
    function isAdmin() {
      return isAuthenticated() && request.auth.token.admin == true;
    }

    // Documents Folder (e.g. GST docs, Aadhaar cards)
    // - Authenticated users can upload documents (max 5MB, image only)
    // - Authenticated users and Admins can view documents
    match /documents/{fileName} {
      allow read: if true;
      allow write: if request.resource.size < 5 * 1024 * 1024 // 5MB max
                   && request.resource.contentType.matches('image/.*'); // Only images
    }

    // Product Images Folder
    // - Anyone can view product images
    // - Only Admins can upload, modify, or delete product images
    match /products/{fileName} {
      allow read: if true;
      allow write: if isAdmin()
                   && request.resource.size < 5 * 1024 * 1024
                   && request.resource.contentType.matches('image/.*');
    }

    // Quotes Folder
    // - Authenticated users (sellers) can upload, modify, delete quotes.
    match /quotes/{fileName} {
      allow read: if true;
      allow create, update: if isAuthenticated()
                   && request.resource.size < 5 * 1024 * 1024
                   && request.resource.contentType == 'application/pdf';
      allow delete: if isAuthenticated();
    }

    // Default fallback - deny all other access
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
